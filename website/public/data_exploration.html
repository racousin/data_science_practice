def data_preprocessing(data, target=None, id_col=None):
    """Prepare data for modeling by handling missing values and separating features from the target and IDs."""
    if id_col:
        IDs = data[id_col]
        data = data.drop(columns=[id_col])
    else:
        IDs = None
    
    if target:
        Y = data[target]
        X = data.drop(columns=[target])
    else:
        Y = None
        X = data
    
    X = X.fillna(0)
    return IDs, X, Y

def train_model(X, Y):
    """Train a machine learning model and save it to a file."""
    model = LinearRegression(n_jobs=-1)
    model.fit(X, Y)
    with open('model.pkl', 'wb') as f:
        pickle.dump(model, f)
    return model
In [20]:
def load_model(file_name='model.pkl'):
    """Load a machine learning model from a file."""
    with open(file_name, 'rb') as f:
        model = pickle.load(f)
    return model

def serving_model(X, model=None):
    """Serve predictions using the trained model."""
    if model is None:
        model = load_model()
    predictions = model.predict(X)
    return predictions

def serve_data(IDs, predictions, filename="predictions.csv"):
    """Write IDs and predictions to a CSV file."""
    results = pd.DataFrame({
        'ID': IDs,
        'Prediction': predictions
    })
    results.to_csv(filename, index=False)
    print(f"Predictions with IDs written to {filename}")
