import pandas as pd
import numpy as np
from datetime import datetime, timedelta
from sklearn.linear_model import LinearRegression

np.random.seed(42)

# Generate 1000 samples
n_samples = 1000

# Generate features
stores = ['StoreA', 'StoreB', 'StoreC', 'StoreD', 'StoreE']

data = {
    'product_id': [f'P{i:04d}' for i in range(1, n_samples + 1)],
    'store': np.random.choice(stores, n_samples),
    'weight': np.round(np.random.uniform(0.1, 10.0, n_samples), 2),
    'length': np.round(np.random.uniform(5, 100, n_samples), 2),
    'width': np.round(np.random.uniform(5, 100, n_samples), 2),
    'height': np.round(np.random.uniform(5, 100, n_samples), 2),
    'rating': np.random.randint(1, 6, n_samples),
    'num_reviews': np.random.randint(0, 1000, n_samples),
    'days_since_last_sale': np.random.randint(0, 365, n_samples),
}

# Create DataFrame
df = pd.DataFrame(data)

# Add some derived features
df['volume'] = df['length'] * df['width'] * df['height']
df['days_in_stock'] = np.random.randint(1, 1000, n_samples)

# Generate price
price = (
    10 +
    0.5 * df['weight'] +
    np.random.normal(0, 5, n_samples)  # Add some noise
)
df['price'] = np.round(np.maximum(price, 1), 2)  # Ensure minimum price of 1

def sales_function(row, store_effect):
    base_sales = (
        20 +
        2 * row['weight'] +
        0.0001 * row['volume'] +
        10 * row['rating'] +
        0.05 * row['num_reviews'] +
        -0.05 * row['days_since_last_sale'] +
        -0.5 * row['price']
    )
    return np.round(np.maximum(base_sales * store_effect + np.random.normal(0, 15), 0)).astype(int)

# Store-specific effects for A-D
store_sales_effects = {
    'StoreA': 2,
    'StoreB': 1.5,
    'StoreC': 0.3,
    'StoreD': 0.5,
    'StoreE': 1  # Placeholder effect for Store E (will be replaced with regression prediction)
}

# Apply the sales function for Stores A-D
df['number_sold'] = df.apply(lambda row: sales_function(row, store_sales_effects[row['store']]), axis=1)

# Add datetime column
start_date = datetime(2023, 1, 1)
df['last_updated'] = [start_date + timedelta(days=x) for x in range(n_samples)]

# Split data for Stores A-D and Store E
df_ad = df[df['store'].isin(['StoreA', 'StoreB', 'StoreC', 'StoreD'])]
df_e = df[df['store'] == 'StoreE']

# Features to use in regression (excluding 'product_id' and 'last_updated')
features = ['weight', 'length', 'width', 'height', 'volume', 'rating', 'num_reviews', 'days_since_last_sale', 'price']

# Train a linear regression model on Stores A-D
X_train = df_ad[features]
y_train = df_ad['number_sold']

model = LinearRegression()
model.fit(X_train, y_train)

# Predict 'number_sold' for Store E using the trained model
X_test = df_e[features]
df.loc[df['store'] == 'StoreE', 'number_sold'] = np.round(np.maximum(model.predict(X_test).astype(int) + np.random.normal(0, 3), 0)).astype(int)

# Display the first few rows and data info
print(df.head())
print(df.info())

# Save the complete dataset
df.to_csv('complete_store_dataset.csv', index=False)

def add_null(df, n = 3, columns = ['weight', 'length', 'width', 'days_since_last_sale', 'days_in_stock', 'price']):
    # Randomly select one column
    selected_column = np.random.choice(columns)
    
    # Randomly select 'n' indices to set to NaN
    indices = np.random.choice(df.index, size=n, replace=False)
    
    # Set the selected indices in the chosen column to NaN
    df.loc[indices, selected_column] = np.nan

    return df
