Handling Unseen Categories: This method implements an OrdinalEncoder for each categorical column in a dataset, ensuring that unseen categories encountered during model testing do not cause errors by assigning them a default encoded value. This approach maintains data consistency between training and testing phases.

from sklearn.linear_model import LogisticRegression
from sklearn.preprocessing import OrdinalEncoder
import numpy as np

def fit_encode(train, test):
    encoders = {}
    for col in train.columns:
        # Create the OrdinalEncoder for each column with handling for unseen categories
        oe = OrdinalEncoder(handle_unknown='use_encoded_value', unknown_value=-1)
        # Fit the encoder on the training data
        train[col] = oe.fit_transform(train[[col]])
        # Store the encoder in a dictionary to use for transforming the test data
        encoders[col] = oe
        # Transform the test data using the fitted encoder
        test[col] = oe.transform(test[[col]])
    return train, test

# Example data setup and usage
# Apply encoding to training and test sets
X_train_encoded, X_test_encoded = fit_encode(X_train.copy(), X_test.copy())

# Initialize and train the RandomForest model
model = LogisticRegression(random_state=42)
model.fit(X_train_encoded, y_train)
y_pred = model.predict(X_test_encoded)
print("Accuracy with correctly applied OrdinalEncoder:", accuracy_score(y_test, y_pred))

# Define the correct order of categories for 'cap-diameter' and 'has-ring'
cap_diameter_order = ['Tiny', 'Small', 'Moderate', 'Medium', 'Fairly Large', 
                      'Large', 'Very Large', 'Huge', 'Gigantic', 'Colossal']

# Setup the ColumnTransformer with the correct category ordering
preprocessor_ordered = ColumnTransformer(
    transformers=[
        ('nom', OneHotEncoder(), ['cap-shape', 'cap-surface', 'has-ring']),
        ('ord', OrdinalEncoder(categories=[cap_diameter_order]), ['cap-diameter'])
    ])

# Create and fit the pipeline
pipeline_ordered = Pipeline([
    ('preprocessor', preprocessor_ordered),
    ('classifier', LogisticRegression(random_state=42))
])
fitted_pipeline = pipeline_ordered.fit(X_train, y_train)
y_pred_ordered = pipeline_ordered.predict(X_test)

# Output the accuracy
print("Accuracy with ordered ordinal encoding:", accuracy_score(y_test, y_pred_ordered))
