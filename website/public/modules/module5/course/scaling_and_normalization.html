# Function to train and evaluate
def train_and_evaluate(X_train, X_test, y_train, y_test):
    model = KNeighborsRegressor(n_neighbors=5)
    model.fit(X_train, y_train)
    y_pred = model.predict(X_test)
    mse = mean_squared_error(y_test, y_pred)
    r2 = r2_score(y_test, y_pred)
    return mse, r2

# Original data
original_mse, original_r2 = train_and_evaluate(X_train, X_test, y_train, y_test)
print(f"Original data - MSE: {original_mse:.4f}, R2: {original_r2:.4f}")

# Min-Max Scaling
scaler = MinMaxScaler()
X_train_minmax = scaler.fit_transform(X_train)
X_test_minmax = scaler.transform(X_test)
minmax_mse, minmax_r2 = train_and_evaluate(X_train_minmax, X_test_minmax, y_train, y_test)
print(f"Min-Max scaling - MSE: {minmax_mse:.4f}, R2: {minmax_r2:.4f}")

# Standard Scaling
scaler = StandardScaler()
X_train_standard = scaler.fit_transform(X_train)
X_test_standard = scaler.transform(X_test)
standard_mse, standard_r2 = train_and_evaluate(X_train_standard, X_test_standard, y_train, y_test)
print(f"Standard scaling - MSE: {standard_mse:.4f}, R2: {standard_r2:.4f}")

# MaxAbs Scaling
scaler = MaxAbsScaler()
X_train_maxabs = scaler.fit_transform(X_train)
X_test_maxabs = scaler.transform(X_test)
maxabs_mse, maxabs_r2 = train_and_evaluate(X_train_maxabs, X_test_maxabs, y_train, y_test)
print(f"MaxAbs scaling - MSE: {maxabs_mse:.4f}, R2: {maxabs_r2:.4f}")

# L2 Normalization
normalizer = Normalizer(norm='l2')
X_train_l2 = normalizer.fit_transform(X_train)
X_test_l2 = normalizer.transform(X_test)
l2_mse, l2_r2 = train_and_evaluate(X_train_l2, X_test_l2, y_train, y_test)
print(f"L2 normalization - MSE: {l2_mse:.4f}, R2: {l2_r2:.4f}")
